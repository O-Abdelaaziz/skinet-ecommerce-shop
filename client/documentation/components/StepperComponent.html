import {Component, Input, OnInit} from '@angular/core';
import {CdkStepper} from "@angular/cdk/stepper";

@Component({
  selector: 'app-stepper',
  templateUrl: './stepper.component.html',
  styleUrls: ['./stepper.component.scss'],
  providers: [{provide: CdkStepper, useExisting: StepperComponent}]
})
export class StepperComponent extends CdkStepper implements OnInit {
  @Input()
  public linearModeSelected: boolean = false;

  ngOnInit(): void {
    this.linear = this.linearModeSelected;
  }

  onClick(index: number) {
    this.selectedIndex = index;
  }
}

<div class="container">
  <header><h2>Step {{selectedIndex + 1}}/{{steps.length}}</h2></header>
  <ul class="nav nav-pills nav-justified">
    <li class="nav-item" *ngFor="let step of steps; let i=index">
      <button [disabled]="true" [class.active]="selectedIndex ===i" class="nav-link py-3 text-uppercase font-weight-bold btn-block" (click)="onClick(i)">
        <ng-container *ngIf="step.stepLabel; else showLabelText" [ngTemplateOutlet]="step.stepLabel.template">
        </ng-container>
        <ng-template #showLabelText>
          {{ step.label }}
        </ng-template>
        <i class="fa ml-2" [ngClass]="{'fa-check  text-success': step.completed == true,'fa-pencil text-warning': !step.completed && selectedIndex ===i}"></i>
      </button>
    </li>
  </ul>

</div>
